    /**
 * <./../nawi/widgets/view/tabview/TabView.js>
 * @file TabView widget static class.
 */

import TabViewProto from "./TabViewProto";
import Widget       from "./../../widget/Widget";

let TabViewConstructor = document.registerElement("nawi-tabview", TabViewProto);

/**
 * TabView widget creator.
 */
export default class TabView
{
  /**
   * Not invokable static constructor.
   */
  constructor()
  {
    throw new Error("[Nawi] 'TabView' class: cannot create an instance of a static class");
  }
  /**
   * Creates TabView widget.
   * @public
   * @param   {String}       id=null         - Widget id.
   * @param   {HTMLElement}  parentNode=null - Parent node.
   * @returns {TabViewProto} New TabView widget node.
   */
  static create(id = null, parentNode = null)
  {
    return Widget.factory(TabViewConstructor, id, parentNode);
  }
}
