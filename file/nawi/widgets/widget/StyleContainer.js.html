    /**
 * <./../nawi/widgets/widget/StyleContatainer.js>
 * @file Style declaration contatiner class.
 */

import { Kernel as std } from "./../core/tools/Kernel";
import NMap              from "./../core/tools/NMap";
//import {resizeEvent} from "./ResizeEvent";

/**
 * Style declaration properties container 
 */
export default class StyleContainer
{
  /**
   * StyleContainer constructor
   * @param {String[]} propertyList=null    - List of style declaration properties
   * @param {Boolean}  preventDefault=false - Default initialization preventing flag
   */
  constructor(propertyList = null, preventDefault = false)
  {
    let defaultProperties = 
    {
      position: "static",
      width   : "auto"  ,
      height  : "auto"  ,
      title   : ""      ,
      group   : null
    };
    
    if(!preventDefault)
    {
      this.generateContainers(defaultProperties);
    }
    
    if(!std.isNull(propertyList))
    {
      this.generateContainers(propertyList);
    }
    
    //this._groupParsing = false;
    
    //this._resizeList = resizeEvent.getEventList();
  }
  /**
   * Generates a widget declaration container
   * @param {Object} propertyList: Declaration property list
   */
  generateContainers(propertyList)
  {
    for(let _prop in propertyList)
    {
      if(!std.isNull(propertyList[_prop]))
      {
        this[_prop] = new NMap([["default", propertyList[_prop]]]);
        this[_prop].set("current", propertyList[_prop]);
      }
      else
      {
        this[_prop] = new NMap();
      }
    }
  }
}
