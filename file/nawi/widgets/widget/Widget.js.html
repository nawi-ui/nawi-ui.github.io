 * <./../nawi/core/widget/Widget.js>
 * @file Widget static class

import { Kernel as std } from "./../../core/tools/Kernel";

/**
 * Base widget logic
 */
export default class Widget
{
  /**
   * Widget static non instantiable constructor
   */
  constructor()
  {
    throw new Error("[Nawi] 'Widget' class: cannot create an instance of a static class");
  }
  /**
   * Creates widget from an input widget constructor
   * @param {WidgetProto} WidgetConstructor Widget constructor
   * @param {String}      id                Widget id
   * @param {HTMLElement} parentNode        Parent node
   */
  static factory(WidgetConstructor, id, parentNode)
  {
    var newWidget = new WidgetConstructor();

    if(!std.isNull(id))
    {
      newWidget.setAttribute("id", id);
    }
    if(!std.isNull(parentNode))
    {
      parentNode.appendChild(newWidget);
    }

    return newWidget;
  }
}
