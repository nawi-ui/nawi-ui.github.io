/**
 * Does deep copy of an object
 * @param {Object} destObj: Destination object
 * @param {Object} srcObj:  Source object
 */
export function copyObject(destObj, srcObj)
{
  var it;

  if(destObj)
  {
    if(!isObject(destObj) || destObj === null)
    {
      throw new Error("[Kt] 'copyObject' function: " + "a destination object '" + destObj.toString() + "' must have an object type");
    }

    for(it in srcObj)
    {
      if(!isObject(srcObj[it]) || srcObj[it] === null)
      {
        destObj[it] = srcObj[it];
      }
      if(isObject(srcObj[it]) && srcObj[it] !== null && srcObj[it].length !== undefined)
      {
        destObj[it] = [];
        copyArray(destObj[it], srcObj[it]);
        continue;
      }
      if(isObject(srcObj[it]) && srcObj[it] !== null)
      {
        destObj[it] = {};
        copyObject(destObj[it], srcObj[it]);
      }
    }	
  }
  else
  {
    throw new Error("[Kt] 'copyObject' function: set a non empty parameter: [object]");
  }
}
/**
 * Does deep copy of an array
 * @param {Object} destObj: Destination array
 * @param {Object} srcObj:  Source array
 */
export function copyArray(destArr, srcArr)
{
  var it;

  if(destArr)
  {
    if(!isObject(destArr) || destArr.length === undefined || destArr === null)
    {
      throw new Error("[Kt] 'copyArray' function: " + "a destination object '" + destArr.toString() + "' must have an array type");
    }

    for(it in srcArr)
    {
      if(!isObject(srcArr[it]) || srcArr[it] === null)
      {
        destArr[it] = srcArr[it];
      }
      if(isObject(srcArr[it]) && srcArr[it] !== null && srcArr[it].length !== undefined)
      {
        destArr[it] = [];
        copyArray(destArr[it], srcArr[it]);
        continue;
      }
      if(isObject(srcArr[it]) && srcArr[it] !== null)
      {
        destArr[it] = {};
        copyObject(destArr[it], srcArr[it]);
      }
    }	
  }
  else
  {
    throw new Error("[Kt] 'copyArray' function: set a non empty parameter: [array]");
  }
}
/**

export function attachClass(element, classList)
{
  let inClassList;
    
  if(isArray(classList))
  {
    inClassList = classList;
  }
  else
  {
    inClassList = [];
    inClassList.push(classList);
  }
  
  for(let i = 0; i < inClassList.length; ++i)
  {
    element.classList.add(inClassList[i]);
  }
}
/**
 * Makes an array from input parameters

 * @return {Array} Output array of parameters
 */
export function makeParametersArray(srcParameters)
{
  let outParameters;
    
  if(isArray(srcParameters))
  {
    outParameters = srcParameters;
  }
  else
  {
    outParameters = [];
    outParameters.push(srcParameters);
  }
  
  return outParameters;
}
/**
 * Tests a readiness to ES6 features using
 * @return {Boolean} readiness
 */
export function isReadyToES6()
{
  return (
          !isUndefined(Array.from)    &&
          !isUndefined(Object.assign) &&
          !isUndefined(window.Map)
         );
}
